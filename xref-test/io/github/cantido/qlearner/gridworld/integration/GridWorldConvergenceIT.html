16  package io.github.cantido.qlearner.gridworld.integration;
17  
18  import static org.apache.commons.math3.stat.StatUtils.percentile;
19  import static org.hamcrest.Matchers.lessThanOrEqualTo;
20  import static org.junit.Assert.assertThat;
21  
22  import io.github.cantido.qlearner.agent.Agent;
23  import io.github.cantido.qlearner.gridworld.client.GridWorldBuilder;
24  import io.github.cantido.qlearner.gridworld.client.GridWorldEnvironment;
25  import org.junit.Rule;
26  import org.junit.Test;
27  import org.junit.rules.Timeout;
28  
29  /**
30   * Tests that the {@link GridWorldEnvironment} implementation of this library converges on the goal
31   * state.
32   */
33  public class GridWorldConvergenceIT {
34  
35    /**
36     * The time limit for these tests.
37     * 
38     * <p>
39     * If there is a problem with the Q-learning algorithm, the values will never converge. That will
40     * cause the tests to run too long. If this test fails due to a timeout, we can be pretty sure
41     * that the algorithm is not converging as it should.
42     * </p>
43     */
44    @Rule
45    public Timeout timeout = Timeout.seconds(1);
46  
47    private final GridWorldBuilder gridWorldBuilder = new GridWorldBuilder();
48  
49    private Agent agent = gridWorldBuilder.getAgent();
50    private GridWorldEnvironment environment =
51        (GridWorldEnvironment) gridWorldBuilder.getEnvironment();
52  
53    /**
54     * We can reasonably expect the agent to learn the environment after this many successful
55     * sessions.
56     */
57    private static final double SUCCESSES_UNTIL_CONVERGENCE = 10;
58  
59    /**
60     * The "correct" solution to our gridworld takes exactly 20 steps. After a learning period, the
61     * agent should take this many steps most of the time.
62     */
63    private static final double EXACT_SOLUTION = 20;
64  
65    /**
66     * How many successes we want to run this test for.
67     */
68    private static final int MAX_SUCCESS_COUNT = 1000;
69  
70    /**
71     * The percentage of successes that have step counts above {@link #EXACT_SOLUTION}.
72     */
73    private static final double CONVERGENCE_PERCENTAGE =
74        SUCCESSES_UNTIL_CONVERGENCE / MAX_SUCCESS_COUNT;
75  
76  
77    /**
78     * Tests that the number of steps that we take to reach the goal goes below an observed value.
79     * <p>
80     * This will run the agent until it reaches the goal a certain number of times, and will calculate
81     * the average number of steps it took to reach the goal state.
82     * </p><p>
83     * There is the potential for the agent to never find the goal state, and for this test to take an
84     * infinite amount of time. For that reason, {@link #timeout} has been set up to guard us from
85     * that.
86     * </p>
87     */
88    @Test
89    public void reachesHighSuccessRate() {
90      // I know the fastest way to do this would be to just divide totalSteps
91      // by totalSuccesses, but creating a double array allows us to do some
92      // more fancy statistics, if I ever write tests for that.
93  
94      double[] stepCounts = new double[MAX_SUCCESS_COUNT];
95  
96      for (int successCount = 0; successCount < MAX_SUCCESS_COUNT; successCount++) {
97        int stepCount = 0;
98  
99        while (!environment.isAtGoalState()) {
100         agent.takeNextAction();
101         stepCount++;
102       }
103 
104       stepCounts[successCount] = stepCount;
105       environment.reset();
106       agent.reset();
107     }
108 
109     assertThat(percentile(stepCounts, CONVERGENCE_PERCENTAGE), lessThanOrEqualTo(EXACT_SOLUTION));
110   }
111 }
