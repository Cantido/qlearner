40  /**
41   * Tests for {@link RandomExplorationStrategy}.
42   */
43  @SuppressWarnings({"null"})
44  public class RandomExplorationStrategyTest {
45    /**
46     * JUnit {@link Rule} to add {@link Mockito} functionality.
47     */
48    @Rule public MockitoRule mockitoRule = MockitoJUnit.rule();
49  
50    @Mock
51    Random random;
52  
53    @Mock
54    State state;
55    @Mock
56    Action goodAction;
57    @Mock
58    Action badAction;
59  
60    Quality goodQuality;
61    Quality badQuality;
62  
63    StateActionQuality goodTriplet;
64    StateActionQuality badTriplet;
65  
66    SortedSet<StateActionQuality> stateActionQualities;
67  
68  
69    /**
70     * Build a {@link StateActionQuality} triplet with a higher {@link Quality} and a lower
71     * {@code Quality}.
72     */
73    @Before
74    public void setUp() {
75      goodQuality = new Quality(1.0);
76      badQuality = new Quality(0.0);
77  
78      goodTriplet = new StateActionQuality(state, goodAction, goodQuality);
79      badTriplet = new StateActionQuality(state, badAction, badQuality);
80  
81      stateActionQualities = new TreeSet<>();
82      stateActionQualities.add(goodTriplet);
83      stateActionQualities.add(badTriplet);
84    }
85  
86    /**
87     * If a {@link RandomExplorationStrategy} has an {@link ExplorationFactor} of zero, then it
88     * should never explore, and always pick the best action.
89     */
90    @Test
91    public void nonexploringStrategyGetsBestAction() {
92      RandomExplorationStrategy strategy =
93          new RandomExplorationStrategy(ExplorationFactor.NEVER_EXPLORE, random);
94  
95      Action actualAction = strategy.getNextAction(stateActionQualities);
96  
97      assertThat(actualAction, is(goodAction));
98    }
99  
100   /**
101    * If a {@link RandomExplorationStrategy} has an {@link ExplorationFactor} of one, then it
102    * should always explore, and it should pick an action randomly.
103    */
104   @Test
105   public void alwaysExploringStrategySelectsRandomAction() {
106     RandomExplorationStrategy strategy =
107         new RandomExplorationStrategy(ExplorationFactor.ALWAYS_EXPLORE, random);
108 
109     Action actualAction = strategy.getNextAction(stateActionQualities);
110 
111     assertThat(actualAction, isOneOf(goodAction, badAction));
112   }
