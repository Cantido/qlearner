34  import java.util.HashSet;

37  @SuppressWarnings({"null", "javadoc"})
38  public class QualityHashMapTest {
39    @Rule
40    public MockitoRule mockito = MockitoJUnit.rule();
41  
42    QualityMap map = new QualityHashMap();
43    @Mock
44    State state;
45    @Mock
46    Action highestAction;
47    @Mock
48    Action middleAction;
49    @Mock
50    Action worstAction;
51    Quality lowestQuality = new Quality(-1.0);
52    Quality middleQuality = new Quality(0.0);
53    Quality highestQuality = new Quality(1.0);
54    Quality defaultQuality = map.getDefaultQuality();
55  
56    Set<Action> actions = new HashSet<>();
57  
58    @Before
59    public void setUp() {
60      actions.add(worstAction);
61      actions.add(middleAction);
62      actions.add(highestAction);
63  
64      when(state.getActions()).thenReturn(actions);
65    }
66  
67    public void fillMap() {
68      map.put(state, highestAction, lowestQuality);
69      map.put(state, middleAction, middleQuality);
70      map.put(state, highestAction, highestQuality);
71    }
72  
73    @Test
74    public void storesQuality() {
75      fillMap();
76  
77      Quality gotQuality = map.get(state, highestAction);
78  
79      assertThat(gotQuality, is(highestQuality));
80    }
81  
82    @Test
83    public void returnsDefaultQualityForBestWhenEmpty() {
84      Quality actualQuality = map.getBestQuality(state);
85  
86      assertThat(actualQuality, is(defaultQuality));
87    }
88  
89    @Test
90    public void returnsDefaultQualityForActionWhenEmpty() {
91      Quality actualQuality = map.get(state, highestAction);
92  
93      assertThat(actualQuality, is(defaultQuality));
94    }
95  
96    @Test
97    public void getsHighestQualityForBest() {
98      fillMap();
99  
100     Quality actualQuality = map.getBestQuality(state);
101 
102     assertThat(actualQuality, is(highestQuality));
103   }
104 
105   @Test
106   public void willNotDeleteDuplicateQualities() {
107     fillMap();
108 
109     map.put(state, highestAction, middleQuality);
110     map.put(state, middleAction, middleQuality);
111 
112     assertThat(map.getBestQuality(state), is(middleQuality));
113 
114     map.put(state, highestAction, highestQuality);
115     map.put(state, middleAction, middleQuality);
116 
117     assertThat(map.getBestQuality(state), is(highestQuality));
118   }
