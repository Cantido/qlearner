32  @SuppressWarnings({"null", "javadoc"})
33  @RunWith(JUnitParamsRunner.class)
34  public class LearningRateTest {
35    @Rule
36    public ExpectedException exception = ExpectedException.none();
37    
38    @Test
39    @Parameters({"-0", "0", "0.1", "0.5", "0.9", "1"})
40    public void learningRateBetweenZeroAndOneInclusiveIsLegal(double learningRate) {
41      // No exception expected
42      new LearningRate(learningRate);
43    }
44  
45    @Test
46    @Parameters({"-0.1", "-1"})
47    public void learningRateLessThanZeroIsIllegal(double learningRate) {
48      exception.expect(IllegalArgumentException.class);
49      new LearningRate(learningRate);
50    }
51  
52    @Test
53    @Parameters({"1.1", "2.0"})
54    public void learningRateGreaterThanOneIsIllegal(double learningRate) {
55      exception.expect(IllegalArgumentException.class);
56      new LearningRate(learningRate);
57    }
58    
59    @Test
60    @Parameters({"-0", "0", "0.1", "0.5", "0.9", "1"})
61    public void convertedNumberValuesAreCorrect(double learningRate) {
62      LearningRate rate = new LearningRate(learningRate);
63      
64      assertThat(rate.intValue(), is(Double.valueOf(learningRate).intValue()));
65      assertThat(rate.longValue(), is(Double.valueOf(learningRate).longValue()));
66      assertThat(rate.floatValue(), is(Double.valueOf(learningRate).floatValue()));
67      assertThat(rate.doubleValue(), is(Double.valueOf(learningRate).doubleValue()));
68    }
69    
70    @Test
71    @Parameters({"-0", "0", "0.1", "0.5", "0.9", "1"})
72    public void toStringContainsValue(double learningRate) {
73      LearningRate rate = new LearningRate(learningRate);
74      assertThat(rate, hasToString(containsString(Double.toString(learningRate))));
75    }
76    
77    @Test
78    public void equalsTest() throws Exception {
79      new EqualsTester()
80            .addEqualityGroup(
81                new LearningRate(1),
82                new LearningRate(1.0),
83                new LearningRate(1L),
84                new LearningRate((float) 1.0))
85            .addEqualityGroup(
86                new LearningRate(0.5),
87                new LearningRate((float) 0.5))
88            .addEqualityGroup(
89                new LearningRate(0),
90                new LearningRate(0.0),
91                new LearningRate((float) 0.0),
92                new LearningRate(0L))
93            .testEquals();
94    }
