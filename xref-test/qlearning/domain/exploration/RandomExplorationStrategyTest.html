62    public void setUp() {
63      goodQuality = new Quality(1.0);
64      badQuality = new Quality(0.0);
65  
66      goodTriplet = new StateActionQuality(state, goodAction, goodQuality);
67      badTriplet = new StateActionQuality(state, badAction, badQuality);
68  
69      stateActionQualities = new TreeSet<>();
70      stateActionQualities.add(goodTriplet);
71      stateActionQualities.add(badTriplet);
72    }
73  
74    @Test
75    public void getBestAction() {
76      RandomExplorationStrategy strategy =
77          new RandomExplorationStrategy(ExplorationFactor.NEVER_EXPLORE, random);
78  
79      Action actualAction = strategy.getNextAction(stateActionQualities);
80  
81      assertThat(actualAction, is(goodAction));
82    }
83  
84    @Test
85    public void getRandomAction() {
86      RandomExplorationStrategy strategy =
87          new RandomExplorationStrategy(ExplorationFactor.ALWAYS_EXPLORE, random);
88  
89      Action actualAction = strategy.getNextAction(stateActionQualities);
90  
91      assertThat(actualAction, isOneOf(goodAction, badAction));
