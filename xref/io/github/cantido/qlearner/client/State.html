16  package io.github.cantido.qlearner.client;

18  import io.github.cantido.qlearner.agent.Agent;
19  import io.github.cantido.qlearner.algorithm.model.Reward;

21  import java.util.Set;

23  import javax.annotation.concurrent.Immutable;
24  import javax.annotation.concurrent.ThreadSafe;
25  
26  /**
27   * A condition that you want to reward or discourage the system from arriving at.
28   * <p>
29   * A {@code State} has possible {@link Action}s that can be taken that are expected to have some
30   * impact on your {@link Environment}. These {@code Action}s are provided by this class's 
31   * {@link #getActions} method. The desirability of this {@code State} is determined by the
32   * {@link Reward} value provided by {@link #getReward}.
33   * </p>
34   * <p>
35   * <strong>Warning</strong>: This class <em>must</em> implement {@link #hashCode()} and
36   * {@link #equals(Object)}!
37   * The system uses these comparisons to look up the quality values of {@code State-Action} pairs, so
38   * if two different {@code State}s are seen as equivalent, then the system will not work correctly.
39   * For that reason, {@code State} objects must be immutable.
40   * </p>
41   */
42  @Immutable
43  @ThreadSafe
44  public abstract class State {
45    /**
46     * Get the desirability of the current state. Higher values will make the {@link Agent} prefer
47     * this state, and lower values will make it avoid this state. It is best to choose rewards based
48     * on the desirability of the outcome, not of behaviors, to allow the system the most freedom to
49     * search. It robs the algorithm of its power to reward for behavior and not outcome.
50     * 
51     * @return the score of this state's desirability
52     */
53    public abstract Reward getReward();
54  
55    /**
56     * Get all {@link Action}s that is possible to perform while in this state.
57     * 
58     * @return all gridworld.actions that can be taken
59     */
60    public abstract Set<Action> getActions();
61  }
