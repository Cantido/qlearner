16  package io.github.cantido.qlearner.agent;
17  
18  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
19  import io.github.cantido.qlearner.client.Action;
20  import io.github.cantido.qlearner.client.State;
21  
22  import java.util.Objects;
23  
24  import javax.annotation.Nonnull;
25  import javax.annotation.Nullable;
26  import javax.annotation.Signed;
27  import javax.annotation.concurrent.Immutable;
28  import javax.annotation.concurrent.ThreadSafe;
29  
30  /**
31   * A single {@link State}-{@link Action} pair choice made by an {@link Agent} during one iteration
32   * of its algorithm.
33   * <p>
34   * The {@code State} represents the state that the agent was in when it started an iteration, and
35   * the {@code Action} represents the action that was chosen to leave that state.
36   * </p>
37   */
38  @Immutable
39  @ThreadSafe
40  public class Step {
41    @Nonnull
42    private final State startingState;
43    @Nonnull
44    private final Action leavingAction;
45  
46    /**
47     * Create a {@code Step} containing the given {@code State} and {@code Action}.
48     * 
49     * @param startingState the {@code State} that the agent originated from.
50     * @param leavingAction the {@code Action} that was taken from that state.
51     */
52    public Step(State startingState, Action leavingAction) {
53      this.startingState = startingState;
54      this.leavingAction = leavingAction;
55    }
56  
57    /**
58     * Get the action that was taken to leave this step's {@link State}.
59     * 
60     * @return the {@code Action} that was taken to leave this {@code Step}'s {@code State}.
61     */
62    public Action getLeavingAction() {
63      return leavingAction;
64    }
65  
66    /**
67     * Get the {@code State} that preceded this step's {@link Action}.
68     * 
69     * @return the starting {@code State}.
70     */
71    public State getStartingState() {
72      return startingState;
73    }
74  
75    @SuppressFBWarnings(value = "NP_METHOD_PARAMETER_TIGHTENS_ANNOTATION",
76        justification = "We are overriding equals, which is defined "
77            + "as @Nullable. This is a false positive.")
78    @Override
79    public boolean equals(@Nullable Object other) {
80      if (this == other) {
81        return true;
82      }
83      if (other == null || getClass() != other.getClass()) {
84        return false;
85      }
86      Step step = (Step) other;
87      return Objects.equals(startingState, step.startingState)
88          && Objects.equals(leavingAction, step.leavingAction);
89    }
90  
91    @Override
92    @Signed
93    public int hashCode() {
94      return Objects.hash(startingState, leavingAction);
95    }
96  }
