41  /**
42   * Builds {@link Agent} objects using the Builder pattern.
43   * 
44   * <pre>
45   * Agent agent = new AgentBuilder().setEnvironment(environment)
46   *     .setExplorationStrategy(EXPLORATION_STRATEGY).setLearningRate(LEARNING_RATE)
47   *     .setDiscountFactor(DISCOUNT_FACTOR).setQualityMap(qualityMap).getAgent();
48   * </pre>
49   */
50  @NotThreadSafe
51  public class AgentBuilder {
52    @Nonnull
53    private static final ExplorationFactor DEFAULT_EXPLORATION_FACTOR = new ExplorationFactor(0.2);
54  
55    @Nonnull
56    private final Environment environment;
57    @Nonnull
58    private ExecutorService actionExecutor = MoreExecutors.newDirectExecutorService();
59    @Nonnull
60    private DiscountFactor discountFactor = new DiscountFactor(1);
61    @Nonnull
62    private ExplorationStrategy explorationStrategy =
63        new RandomExplorationStrategy(DEFAULT_EXPLORATION_FACTOR);
64    @Nonnull
65    private LearningRate learningRate = new LearningRate(1);
66    @Nonnull
67    private QualityMap qualityMap = new QualityHashMap();
68    @Nonnull
69    private QualityUpdateStrategy qualityUpdateStrategy = new BackwardInduction();
70    @Nonnull
71    private QualityUpdater qualityUpdater =
72        new QualityUpdater(qualityMap, qualityUpdateStrategy, learningRate, discountFactor);
73  
74    /**
75     * Start building an {@link Agent} that will interact with the given {@link Environment}. You can
76     * immediately call {@link #getAgent()} to get an {@code Agent} populated with default values.
77     * 
78     * @param environment the environment that the result agent will interact with.
79     */
80    public AgentBuilder(Environment environment) {
81      this.environment = environment;
82    }
83  
84    /**
85     * Get the resulting {@link Agent} that this object is building. Subsequent calls will return new
86     * and distinct agent objects.
87     * 
88     * @return the {@code Agent} that this object was building.
89     */
90    public Agent getAgent() {
91      return new Agent(environment, explorationStrategy, qualityMap, actionExecutor, qualityUpdater);
92    }

164   /**
165    * Set the {@code DiscountFactor} that this builder will give to its resulting {@link Agent} to
166    * control how it learns.
167    * 
168    * @param discountFactor the {@code DiscountFactor} that the resulting {@code Agent} will use.
169    * @return this builder, for chaining
170    */
171   public AgentBuilder setDiscountFactor(DiscountFactor discountFactor) {
172     this.discountFactor = discountFactor;
173     this.qualityUpdater = new QualityUpdater(this.qualityMap, this.qualityUpdateStrategy,
174         this.learningRate, discountFactor);
175     return this;
176   }

178   /**
179    * Set the {@code ExecutorService} that this builder will give to its resulting {@link Agent} to
180    * execute chosen {@link Action}s. By default, {@code Agent}s are built with an
181    * {@code ExecutorService} that executes actions in the same thread, also known as a "direct"
182    * executor.
183    * <p>
184    * If your {@code Action}s are fairly lightweight, any thread-based {@code ExecutorService} will
185    * incur too much overhead to be worthwhile. For example using a
186    * {@link Executors#newSingleThreadExecutor()} in the Gridworld example increases its execution
187    * time six-fold, purely through its overhead. Be sure that your {@code Action}s really do need to
188    * be executed in their own threads.
189    * </p>
190    * 
191    * @param executor the {@code ExecutorService} that the resulting {@code Agent} will use.
192    * @return this builder, for chaining
193    */
194   public AgentBuilder setExecutorService(ExecutorService executor) {
195     this.actionExecutor = executor;
196     return this;
197   }
