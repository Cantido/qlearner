16  package io.github.cantido.qlearner.algorithm.model;
17  
18  import javax.annotation.CheckForSigned;
19  import javax.annotation.Nonnegative;
20  import javax.annotation.Nonnull;
21  import javax.annotation.Signed;
22  import javax.annotation.concurrent.Immutable;
23  import javax.annotation.concurrent.ThreadSafe;
24  
25  /**
26   * The discount factor determines the importance of future rewards.
27   * 
28   * <p>
29   * A factor of 0 will make the agent "myopic" (or short-sighted) by only considering current
30   * rewards, while a factor approaching 1 will make it strive for a long-term high reward. If the
31   * discount factor meets or exceeds 1, the quality values may diverge.
32   * </p>
33   */
34  @Immutable
35  @ThreadSafe
36  public final class DiscountFactor extends Number {
37    @Signed
38    private static final long serialVersionUID = 743132734230321555L;
39    @Nonnegative
40    private static final double ZERO = 0.0;
41  
42    @Nonnull
43    @Nonnegative
44    private final Double value;
45  
46    /**
47     * Create a new {@code DiscountFactor} with the given value.
48     * 
49     * @param value the value of the new discount factor, in the range [0, &infin;)
50     */
51    public DiscountFactor(@CheckForSigned @Nonnegative Number value) {
52      if (value.doubleValue() < ZERO) {
53        throw new IllegalArgumentException(
54            "Discount factor must be greater than or equal to zero, but got: " + value);
55      }
56  
57      this.value = value.doubleValue();
58    }
59  
60    @Override
61    @Nonnegative
62    public int intValue() {
63      return value.intValue();
64    }
65  
66    @Override
67    @Nonnegative
68    public long longValue() {
69      return value.longValue();
70    }
71  
72    @Override
73    @Nonnegative
74    public float floatValue() {
75      return value.floatValue();
76    }
77  
78    @Override
79    @Nonnegative
80    public double doubleValue() {
81      return value;
82    }
83  
84    @Override
85    public String toString() {
86      return "DiscountFactor[" + value + "]";
87    }
88  }
