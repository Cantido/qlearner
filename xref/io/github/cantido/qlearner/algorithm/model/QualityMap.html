16  package io.github.cantido.qlearner.algorithm.model;
17  
18  import io.github.cantido.qlearner.algorithm.quality.StateActionQuality;
19  import io.github.cantido.qlearner.client.Action;
20  import io.github.cantido.qlearner.client.State;
21  
22  /**
23   * An object that can store the {@link Quality} values of {@link State}-{@link Action} pairs.
24   */
25  public interface QualityMap {
26    /**
27     * Store the specified value.
28     * 
29     * @param state the starting {@code State}.
30     * @param action the {@code Action} taken from that {@code State}.
31     * @param quality the {@code Quality} of that combination.
32     */
33    void put(State state, Action action, Quality quality);
34  
35    /**
36     * Retrieve a value from this object.
37     * 
38     * @param state a starting {@code State}.
39     * @param action the {@code Action} taken from that {@code State}.
40     * @return the {@code Quality} of that combination, or the specified default value if the given
41     *         {@code State}-{@code Action} combination has not yet been stored.
42     * 
43     * @see #getDefaultQuality()
44     */
45    Quality get(State state, Action action);
46  
47    /**
48     * Get the best possible quality for the given {@code State}.
49     * 
50     * @param state the {@code State} to find the best {@code Quality} for
51     * @return the highest {@code Quality} for any {@code Action} that can be taken from the specified
52     *         {@code State}.
53     */
54    Quality getBestQuality(State state);
55  
56    /**
57     * Get the {@code Quality} vlaue that will be returned for any {@code State}-{@code Action}
58     * combination has not yet been stored.
59     * 
60     * @return the default {@code Quality} value.
61     */
62    Quality getDefaultQuality();
63  
64    /**
65     * Build a {@code StateActionQuality} triplet for the specified {@code State}, {@code Action}, and
66     * that combination's {@code Quality} according to this object.
67     * 
68     * @param state the {@code State} for the given triplet.
69     * @param action the {@code Action} for the given triplet.
70     * @return a triplet containing the pair and their {@code Quality}.
71     */
72    StateActionQuality getTriplet(State state, Action action);
73  }
