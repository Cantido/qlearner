18  import io.github.cantido.qlearner.algorithm.model.DiscountFactor;
19  import io.github.cantido.qlearner.algorithm.model.LearningRate;
20  import io.github.cantido.qlearner.algorithm.model.Quality;
21  import io.github.cantido.qlearner.algorithm.model.QualityUpdateStrategy;
22  import io.github.cantido.qlearner.algorithm.model.Reward;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;

26  import javax.annotation.Nonnull;
27  import javax.annotation.Signed;
28  import javax.annotation.concurrent.Immutable;
29  import javax.annotation.concurrent.ThreadSafe;

31  /**
32   * A strategy for updating {@link Quality} values by looking at backwards at the {@link Reward}s
33   * accumulated thus far.
34   */
35  @Immutable
36  @ThreadSafe
37  public class BackwardInduction implements QualityUpdateStrategy {
38    @SuppressWarnings("null")
39    @Nonnull
40    private static final Logger logger = LoggerFactory.getLogger(BackwardInduction.class);
41  
42    @Override
43    public Quality next(
44                        Quality oldQuality,
45                        LearningRate learningRate,
46                        Reward reward,
47                        DiscountFactor discountFactor,
48                        Quality optimalFutureValueEstimate) {
49  
50      logger.debug(
51          "Creating new quality using the following values: "
52              + "(Qt: {}), (a: {}), (Rt+1: {}), (d: {}), (maxQt: {})",
53          oldQuality, learningRate, reward, discountFactor, optimalFutureValueEstimate);
54  
55      double oldQualityValue = oldQuality.doubleValue();
56  
57      @Signed
58      double qualityValue =
59        oldQualityValue
60          + (learningRate.doubleValue()
61              * (reward.doubleValue()
62                  + discountFactor.doubleValue() * optimalFutureValueEstimate.doubleValue()
63                  - oldQualityValue));
64  
65      return new Quality(qualityValue);
66    }
67  }
