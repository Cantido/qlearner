16  package io.github.cantido.qlearner.algorithm.quality;

18  import io.github.cantido.qlearner.algorithm.model.Quality;
19  import io.github.cantido.qlearner.algorithm.model.QualityMap;
20  import io.github.cantido.qlearner.client.Action;
21  import io.github.cantido.qlearner.client.State;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import java.util.HashMap;
26  import java.util.Map;
27  import java.util.PriorityQueue;
28  
29  import javax.annotation.Nonnegative;
30  import javax.annotation.Nonnull;
31  import javax.annotation.concurrent.NotThreadSafe;
32  
33  /**
34   * A data structure that stores {@link Quality} values in a hashed form.
35   */
36  @NotThreadSafe
37  public class QualityHashMap implements QualityMap {
38    @SuppressWarnings("null")
39    @Nonnull
40    private static final Logger logger = LoggerFactory.getLogger(QualityHashMap.class);
41    @Nonnegative
42    private final int expectedAverageActionsPerState;
43  
44    /**
45     * Mapping of State-Action pairs to their Quality value.
46     */
47    @Nonnull
48    private final Map<State, Map<Action, Quality>> actionQualities;
49    
50    /**
51     * An optimization; stores the best quality for each state.
52     * 
53     * <p>Be careful if you are changing this data structure: it is possible to have two equivalent
54     * quality values for a given state, and then when we update any of the gridworld.actions with
55     * that quality, we would lose that quality for all gridworld.actions.</p>
56     */
57    @Nonnull
58    private final Map<State, PriorityQueue<Quality>> bestQualities;
59    @SuppressWarnings("null")
60    @Nonnull
61    private Quality defaultQuality = Quality.ZERO;
62  
63    /**
64     * Constructs an empty {@code QualityHashMap} with a default number of expected total states (16)
65     * and a default number of expected actions per state (11).
66     */
67    public QualityHashMap() {
68      // Will just match the default PriorityQueue size
69      expectedAverageActionsPerState = 11;
70      actionQualities = new HashMap<>();
71      bestQualities = new HashMap<>();
72    }
73  
74    /**
75     * Constructs an empty {@code QualityHashMap} with a specified number of expected total states and
76     * number of expected actions per state.
77     * 
78     * <p>
79     * Specifying these values is only a performance optimization. It does not change the
80     * functionality of this object.
81     * </p>
82     * 
83     * @param expectedStates the approximate total count of states this object will store.
84     * @param actionsPerState the average count of actions that each stored state will have.
85     */
86    public QualityHashMap(@Nonnegative int expectedStates, @Nonnegative int actionsPerState) {
87      if (expectedStates < 0) {
88        throw new IllegalArgumentException(
89            "Was given a negative expectedStates number, which is invalid. Got: " + expectedStates);
90      }
91      if (actionsPerState < 0) {
92        throw new IllegalArgumentException(
93            "Was given a negative actionsPerState number, which is invalid. Got: " + actionsPerState);
94      }
95  
96      expectedAverageActionsPerState = actionsPerState;
97      actionQualities = new HashMap<>(expectedStates * actionsPerState);
98      bestQualities = new HashMap<>(expectedStates);
99    }
100 
101   /**
102    * Set the {@code Quality} value that will be returned for {@code State}-{@code Action} pairs that
103    * this object has not yet stored.
104    * 
105    * @param defaultQuality the default {@code Quality} value.
106    */
107   public void setDefaultQuality(Quality defaultQuality) {
108     this.defaultQuality = defaultQuality;
109   }
110 
111   @Override
112   public Quality getDefaultQuality() {
113     return this.defaultQuality;
114   }
115 
116   @Override
117   public void put(State state, Action action, Quality quality) {
118     Quality oldQuality = get(state, action);
119 
120     if (!actionQualities.containsKey(state)) {
121       actionQualities.put(state, new HashMap<>(expectedAverageActionsPerState));
122     }
123     actionQualities.get(state).put(action, quality);
124     //actionQualities.put(new ImmutablePair<>(state, action), quality);
125 
126     PriorityQueue<Quality> queueToUpdate;
127 
128     if (bestQualities.containsKey(state)) {
129       queueToUpdate = bestQualities.get(state);
130       queueToUpdate.remove(oldQuality);
131     } else {
132       queueToUpdate =
133           new PriorityQueue<>(expectedAverageActionsPerState, Quality.DESCENDING_ORDER);
134       bestQualities.put(state, queueToUpdate);
135     }
136 
137     logger.debug("Updating quality for [{}, {}] to {}", state, action, quality);
138 
139     queueToUpdate.add(quality);
140   }
141 
142   @Override
143   public Quality get(State state, Action action) {
144     Map<Action, Quality> actions = actionQualities.get(state);
145     if (actions == null) {
146       return defaultQuality;
147     }
148     
149     Quality quality = actions.get(action);
150 
151     if (quality == null) {
152       return defaultQuality;
153     }
154 
155     return quality;
156   }
157 
158   @Override
159   public Quality getBestQuality(State state) {
160     Quality bestQuality = defaultQuality;
161 
162     if (bestQualities.containsKey(state)) {
163       bestQuality = bestQualities.get(state).peek();
164     }
165     if (bestQuality == null) {
166       return defaultQuality;
167     }
168 
169     return bestQuality;
170   }
171 
172   @Override
173   public StateActionQuality getTriplet(State state, Action action) {
174     return new StateActionQuality(state, action, this.get(state, action));
175   }
