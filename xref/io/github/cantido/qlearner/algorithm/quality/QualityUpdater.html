16  package io.github.cantido.qlearner.algorithm.quality;

19  import io.github.cantido.qlearner.algorithm.model.DiscountFactor;
20  import io.github.cantido.qlearner.algorithm.model.LearningRate;
21  import io.github.cantido.qlearner.algorithm.model.Quality;

26  import io.github.cantido.qlearner.client.State;

28  import javax.annotation.Nonnull;

30  /**
31   * Caches values that are needed by a {@link QualityUpdateStrategy}, but that do not change, to
32   * simplify that strategy's interface.
33   */
34  public class QualityUpdater {
35    @Nonnull
36    private final QualityMap qualityMap;
37    @Nonnull
38    private final QualityUpdateStrategy strategy;
39    @Nonnull
40    private final LearningRate learningRate;
41    @Nonnull
42    private final DiscountFactor discountFactor;
43  
44    /**
45     * Create a new {@code QualityUpdater}.
46     * 
47     * @param qualityMap the map to retrieve values from.
48     * @param strategy the strategy to actually update values with.
49     * @param learningRate the learning rate to pass to the strategy.
50     * @param discountFactor the discount factor to pass to the strategy.
51     */
52    public QualityUpdater(QualityMap qualityMap, QualityUpdateStrategy strategy,
53        LearningRate learningRate, DiscountFactor discountFactor) {
54      this.qualityMap = qualityMap;
55      this.strategy = strategy;
56      this.learningRate = learningRate;
57      this.discountFactor = discountFactor;
58    }
59  
60    /**
61     * Update {@code Quality} values given a starting state, taken action, and the resulting state.
62     * 
63     * @param stepTaken the starting state and action taken.
64     * @param resultingState the final state.
65     */
66    public void updateQuality(Step stepTaken, State resultingState) {
67      State previousState = stepTaken.getStartingState();
68      Action actionTaken = stepTaken.getLeavingAction();
69  
70      Quality oldQuality = qualityMap.get(previousState, actionTaken);
71  
72      Reward reward = resultingState.getReward();
73      Quality optimalFutureValueEstimate = qualityMap.getBestQuality(resultingState);
74  
75      Quality newQuality = this.strategy.next(oldQuality, learningRate, reward, discountFactor,
76          optimalFutureValueEstimate);
77  
78      qualityMap.put(previousState, actionTaken, newQuality);
79    }
80  }
