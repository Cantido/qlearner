16  package io.github.cantido.qlearner.algorithm.exploration;
17  
18  import io.github.cantido.qlearner.algorithm.model.ExplorationFactor;
19  import io.github.cantido.qlearner.algorithm.model.ExplorationStrategy;
20  import io.github.cantido.qlearner.algorithm.quality.StateActionQuality;
21  import io.github.cantido.qlearner.client.Action;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import java.util.Random;
26  import java.util.SortedSet;
27  import java.util.concurrent.ThreadLocalRandom;
28  
29  import javax.annotation.Nonnull;
30  import javax.annotation.concurrent.Immutable;
31  import javax.annotation.concurrent.ThreadSafe;
32  
33  /**
34   * A simple {@link ExplorationStrategy} which will choose a random {@link Action} some of the time.
35   */
36  @Immutable
37  @ThreadSafe
38  public class RandomExplorationStrategy implements ExplorationStrategy {
39    @SuppressWarnings("null")
40    @Nonnull
41    private final Logger logger = LoggerFactory.getLogger(getClass());
42  
43    @Nonnull
44    private final ExplorationFactor explorationFactor;
45    @Nonnull
46    private final Random random;
47  
48    /**
49     * Create a new {@link RandomExplorationStrategy} with the given exploration factor.
50     * 
51     * @param explorationFactor the exploration factor to set
52     */
53    @SuppressWarnings("null")
54    public RandomExplorationStrategy(ExplorationFactor explorationFactor) {
55      this(explorationFactor, ThreadLocalRandom.current());
56    }
57  
58    /**
59     * Create a {@code RandomExplorationStrategy} that uses the given exploration factor and source of
60     * randomness to determine whether or not to explore.
61     * 
62     * @param explorationFactor this object's propensity to explore
63     * @param random the source of randomness to make exploration decisions with
64     */
65    public RandomExplorationStrategy(ExplorationFactor explorationFactor, Random random) {
66      this.explorationFactor = explorationFactor;
67      this.random = random;
68    }
69  
70    /**
71     * Get the exploration factor, which determines how often the agent will choose random
72     * gridworld.actions over desirable ones in order to explore the problem space.
73     * 
74     * @return this strategy's propensity to explore.
75     */
76    public ExplorationFactor getExplorationFactor() {
77      return this.explorationFactor;
78    }
79  
80    @Override
81    public Action getNextAction(SortedSet<StateActionQuality> stateActionQualities) {
82  
83      Action nextAction;
84  
85      Double checkValue = random.nextDouble();
86  
87      if (explorationFactor.shouldExplore(checkValue)) {
88        nextAction = getRandomAction(stateActionQualities);
89        logger.debug("Should explore, chose random action: {}", nextAction);
90      } else {
91        nextAction = getBestAction(stateActionQualities);
92        logger.debug("Should not explore, chose best action: {}", nextAction);
93      }
94  
95      return nextAction;
96    }
97  
98  
99    private Action getBestAction(SortedSet<StateActionQuality> stateActionQualities) {
100     assert stateActionQualities != null : "stateActionQualities must not be null";
101 
102     logger.debug("Determining best action from possible gridworld.actions: {}",
103         stateActionQualities);
104     return stateActionQualities.last().getAction();
105   }
106 
107   private Action getRandomAction(SortedSet<StateActionQuality> stateActionQualities) {
108     assert stateActionQualities != null : "stateActionQualities must not be null";
109 
110     Integer randomIndex = random.nextInt(stateActionQualities.size());
111     StateActionQuality randomTriplet =
112         (StateActionQuality) stateActionQualities.toArray()[randomIndex];
113 
114     return randomTriplet.getAction();
115   }
116 }
